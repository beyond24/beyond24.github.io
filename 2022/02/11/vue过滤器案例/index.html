    vue过滤器案例

    前端 vue 过滤器
    发布时间 : 2022-02-11 15:14
    阅读 :

    实现如下功能：

    根据传入date_string类型的时间数据计算出，距离当前时间的差值

    表达方式为：

    • xx秒前
    • xx天前
    • …
    <!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
    <script src="vue.js"></script>
</head>
<body>
<div id="app">
    <div>
        {{ now|get_date }}
    </div>

    <ul>
        <li v-for="item in data_list" :key="item.id">
            id: {{ item.id }} 时间：{{ item.date|get_date }}
        </li>

    </ul>
        <ul>
        <li v-for="item in data_list" :key="item.id">
            id: {{ item.id }} 时间：{{ item.date }} 发表于:{{ item.date|time_to_filter }}
        </li>

    </ul>
</div>

<script>
    function getDateDiff(datestamp) {
        var publishTime = datestamp / 1000,
            d_seconds,
            d_minutes,
            d_hours,
            d_days,
            timeNow = parseInt(new Date().getTime() / 1000),
            d,
            date = new Date(publishTime * 1000),
            Y = date.getFullYear(),
            M = date.getMonth(),
            D = date.getDate(),
            H = date.getHours(),
            m = date.getMinutes(),
            s = date.getSeconds();
        // 小于10的在前面补0
        if (M < 10) {
            M = '0' + M;
        }
        if (D < 10) {
            D = '0' + D;
        }
        if (H < 10) {
            H = '0' + H;
        }
        if (m < 10) {
            m = '0' + m;
        }
        if (s < 10) {
            s = '0' + s;
        }
        d = timeNow - publishTime;
        d_days = parseInt(d / 86400);
        d_hours = parseInt(d / 3600);
        d_minutes = parseInt(d / 60);
        d_seconds = parseInt(d);
        if (d_days > 0 && d_days < 3) {
            return d_days + "天前";
        } else if (d_days <= 0 && d_minutes > 0) {
            return d_hours + '小时前';
        } else if (d_hours <= 0 && d_hours > 0) {
            return d_minutes + '分钟前';
        } else if (d_seconds < 60) {
            if (d_seconds <= 0) {
                return '刚刚发表';
            } else {
                return d_seconds + '秒前';
            }
        } else if (d_days >= 3 && d_days < 30) {
            return M + '-' + D + ' ' + H + ':' + m;
        }else if (d_days>=0) {
            return Y + '-' + M + '-' + D + ' ' + H + ':' + m;
        }

    }




    new Vue({
        el: '#app', //vue挂载的位置，除了body
        data: {
            now: '2022-2-8 22:21:11',
            data_list: [
                {id: 1, date: '2022-2-8 22:21:11'},
                {id: 2, date: '2022-2-11 9:21:11'},
                {id: 3, date: '2022-2-9 22:21:11'},
                {id: 4, date: '2022-2-10 22:21:11'},
                {id: 5, date: '2022-2-11 11:09:11'},
            ]
        },
        filters: {
            get_date: function (date_str) {
                let date = new Date(date_str)
                return `${date.getFullYear()}-${date.getMonth()}-${date.getDate()}`
            },
            time_to_filter: function (date_str) {
                // 字符串->对象->时间戳
                let datestamp = new Date(date_str).getTime()
                // 和当前时间做差，求出相差多少天，多少秒
                // 字符串->时间戳
                return getDateDiff(datestamp)

            }
        }

    })
</script>
</body>
</html>
