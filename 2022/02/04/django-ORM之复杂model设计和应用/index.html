    django ORM之复杂model设计和应用

    django 数据库 ORM
    发布时间 : 2022-02-04 18:08
    阅读 :

    现给出如下表结构对应关系，完成对应操作

    model.py

    class School(models.Model):
    nid = models.AutoField(primary_key=True)
    title = models.CharField(max_length=16, verbose_name='校区名称')


class Teacher(models.Model):
    nid = models.AutoField(primary_key=True)
    name = models.CharField(max_length=8, verbose_name='教师姓名')
    age = models.IntegerField(verbose_name='教师年龄')
    level = models.CharField(max_length=8, verbose_name='教师等级', null=True)

    def __str__(self):
        return self.name


class Classes(models.Model):
    nid = models.AutoField(primary_key=True)
    title = models.CharField(verbose_name='班级名称', max_length=12)
    teacher = models.ManyToManyField(
        to='Teacher'
    )
    school = models.ForeignKey(
        to='School',  # 班级关联的校区
        to_field='nid',
        on_delete=models.CASCADE,
        null=True,
    )


class Course(models.Model):
    nid = models.AutoField(primary_key=True)
    title = models.CharField(max_length=8, verbose_name='课程名称')


class Student(models.Model):
    nid = models.AutoField(primary_key=True)
    name = models.CharField(max_length=8, verbose_name='学生姓名')
    age = models.IntegerField(verbose_name='学生年龄')
    gender = models.CharField(max_length=1, verbose_name='学生性别', default='男')
    addr = models.CharField(max_length=32, verbose_name='学生地址', null=True, blank=True)
    student_info = models.OneToOneField(
        to='StudentInfo',
        to_field='nid',
        on_delete=models.CASCADE,
        null=True,
    )
    classes = models.ForeignKey(
        to='Classes',
        to_field='nid',
        on_delete=models.CASCADE,
        null=True,
    )
    course = models.ManyToManyField(
        to='Course'  # 课程
    )

    def __str__(self):
        return self.name + str(self.nid)


class StudentInfo(models.Model):
    nid = models.AutoField(primary_key=True)
    position = models.CharField(max_length=8, verbose_name='担任职位', null=True)
    like = models.CharField(max_length=8, verbose_name='学生爱好')
    blood = models.CharField(max_length=8, verbose_name='学生血型')
    email = models.CharField(max_length=32, verbose_name='学生邮箱')
    create_time = models.DateTimeField(verbose_name='学生报名时间', auto_now=True)

    from faker import Faker


faker = Faker('zh_CN')
student_list = []
student_info_list = []
for i in range(10):

    完成如下操作

    # 查询wang的邮箱
# 正向
print(Student.objects.get(name='wang').student_info.email)
# 反向
print(StudentInfo.objects.get(student__name='wang').email)
# 查询每个班级的班级名称
print(Classes.objects.all().values('title'))

# 查询三年一班的所有学生
print(Student.objects.filter(classes__title='三年一班').values('name'))
# 查询新校区的所有学生
print(Classes.objects.filter(school__title='新校区').values('student__name'))
# 或者
print(Student.objects.filter(classes__school__title='新校区').values('name'))
# 查询旧校区职位是班长的学生
print(Student.objects.filter(student_info__position='班长', classes__school__title='旧校区').values('name'))

# 查询选了数学这门课程的学生
print(Student.objects.filter(course__title='数学').values('name'))
# 为还没有职位的同学分配一个职位（学生）
StudentInfo.objects.filter(position__isnull=True).update(position='学生')
