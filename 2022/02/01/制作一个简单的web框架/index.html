    制作一个简单的web框架

    发布时间 : 2022-02-01 22:32
    阅读 :
    1. 现在让我们来开发一个简单的web框架，目录如下

    自己动手写一个简单的web框架，通过此案例，将会对python web的运行模式更为了解

    先使用熟悉的socket实现路由分发

    import socket

sock = socket.socket()
sock.bind(('127.0.0.1', 8000))
sock.listen(5)
# ---------------------------------------
def index():

    server.sendall(b'HTTP /1.1 200 OK \r\n\r\n index_page')


while True:
    server, addr = sock.accept()
    # 1024表示，最多每次接受1024字节
    data =  server.recv(1024)
    # 从request中切片出URL路由-->request:b'GET /index/ HTTP/1.1\r\nHost: 127.0.0.1:8000\r\nConnection: keep-alive\r ... 
    path = data.decode('utf8').split()[1] # split默认在空格出切片
    
    url = {
        '/index/': index,
    }
    view = url.get(path)
    # 若成功获取view，view是一个函数对象
    if view:
        view()
    else:
        server.sendall(b'HTTP /1.1 200 OK \r\n\r\n 404 error')
    server.close()

    使用wsgire模块实现简单的WSGI HTTP Server

    from wsgiref.simple_server import make_server


def app(request, response):
    path = request.get('PATH_INFO')
    # print(path)
    response('200 OK', [('Content-Type', 'text/html')])
    return [b'hello world']




httped = make_server('127.0.0.1', 8000, app)

httped.serve_forever()

    现在让我们来开发一个简单的web框架，目录如下

    # url.py
from view import *


url = {
    '/index': index,
    '/login': login,
    '/register': register,
}

#---------------------------------
# view.py
def index(request):

    return 'index page'

def login(request):

    return '登录 page'
# 虽然以utf8编码传入，但浏览器默认以gbk解码   :鐧诲綍 page
# 需要在response处传入 charset=utf-8 即可

def register(request):

    return render(request, 'register.html')

import os
def render(request, path):

    fp = open(os.path.join('templates', path), 'r', encoding='utf8')
    data = fp.read()
    fp.close()
    return data
#---------------------------------
# manage.py
from wsgiref.simple_server import make_server
from url import url

def app(request, response):
    response('200 OK', [('Content-Type', 'text/html; charset=utf-8')])
    path = request.get('PATH_INFO')


    view = url.get(path)
    if view:
        return [view(request).encode()]
    else:
        return [b'404 error']

httped = make_server('127.0.0.1', 8000, app)

print('Web site is running on http://127.0.0.1:8000')

httped.serve_forever()

    以register视图函数为例

    index和login其实就相当于django中的HttpRsponse类型

    而render也一样可传入request和模板html

    当然也可以将render函数迁移创建一个shortcuts的模块，就更为熟悉了。
